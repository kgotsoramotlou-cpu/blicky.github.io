<script>
const envelope = document.getElementById("envelope");
const overlay = document.getElementById("overlay");
const letter1 = document.getElementById("letter1");
const letter2 = document.getElementById("letter2");
const continueBtn = document.getElementById("continueBtn");
const yesBtn = document.getElementById("yesBtn");
const noBtn = document.getElementById("noBtn");
const celebration = document.getElementById("celebration");
const main = document.getElementById("main");
const typingEl = document.getElementById("typing");
const questionEl = document.getElementById("question");
const choicesEl = document.getElementById("choices");

let noClicks = 0;
let yesScale = 1;
let noScale = 1;

/* OPEN ENVELOPE */
envelope.onclick = () => {
  envelope.classList.add("open");
  setTimeout(() => {
    overlay.style.display = "block";
    letter1.style.display = "block";
  }, 700);
};

/* CONTINUE */
continueBtn.onclick = () => {
  letter1.style.display = "none";
  letter2.style.display = "block";
  startTyping();
};

/* TYPING EFFECT */
const message = "So with that I ask you...";
let index = 0;

function startTyping() {
  typingEl.textContent = "";
  index = 0;
  const interval = setInterval(() => {
    typingEl.textContent += message[index];
    index++;
    if (index === message.length) {
      clearInterval(interval);
      setTimeout(() => {
        questionEl.style.display = "block";
        choicesEl.style.display = "flex";
      }, 500);
    }
  }, 50);
}

/* YES BUTTON */
yesBtn.onclick = celebrate;

/* NO BUTTON LOGIC */
noBtn.onclick = () => {
  noClicks++;

  noBtn.classList.add("shake");
  setTimeout(() => noBtn.classList.remove("shake"), 300);

  if (noClicks <= 10) {
    yesScale += 0.12;
    noScale -= 0.07;

    yesBtn.style.transform = `scale(${yesScale})`;
    noBtn.style.transform = `scale(${noScale})`;

    const x = Math.random() * 200 - 100;
    const y = Math.random() * 200 - 100;
    noBtn.style.translate = `${x}px ${y}px`;
  }

  if (noClicks === 11) {
    noBtn.textContent = "Yes üíñ";
    noBtn.style.background = "#f06292";
    noBtn.onclick = celebrate;
  }
};

/* CELEBRATION */
function celebrate() {
  overlay.style.display = "none";
  letter2.style.display = "none";
  main.style.display = "none";
  celebration.style.display = "flex";
  spawnConfetti();
  heartExplosion();
}

/* FLOATING HEARTS */
setInterval(() => {
  const heart = document.createElement("div");
  heart.className = "heart";
  heart.style.left = Math.random() * 100 + "vw";
  heart.style.background = Math.random() > 0.5 ? "hotpink" : "red";
  heart.style.animationDuration = 6 + Math.random() * 3 + "s";
  heart.style.transform = `rotate(45deg) scale(${0.8 + Math.random() * 0.6})`;

  document.body.appendChild(heart);
  setTimeout(() => heart.remove(), 10000);
}, 220);

/* CONFETTI */
function spawnConfetti() {
  for (let i = 0; i < 250; i++) {
    const confetti = document.createElement("div");
    confetti.className = "confetti";
    confetti.style.left = Math.random() * 100 + "vw";
    confetti.style.background =
      ["#ff69b4", "#ff1493", "#fff", "#ffd1dc", "#ff5fa2"]
      [Math.floor(Math.random() * 5)];
    confetti.style.animationDuration = 2 + Math.random() * 4 + "s";
    confetti.style.transform = `rotate(${Math.random() * 360}deg)`;

    document.body.appendChild(confetti);
    setTimeout(() => confetti.remove(), 7000);
  }
}

/* HEART EXPLOSION */
function heartExplosion() {
  for (let i = 0; i < 40; i++) {
    const h = document.createElement("div");
    h.className = "burst-heart";
    h.style.left = "50%";
    h.style.top = "50%";
    h.style.background = Math.random() > 0.5 ? "red" : "pink";
    h.style.setProperty("--x", (Math.random() * 400 - 200) + "px");
    h.style.setProperty("--y", (Math.random() * 400 - 200) + "px");
    document.body.appendChild(h);
    setTimeout(() => h.remove(), 1500);
  }
}
